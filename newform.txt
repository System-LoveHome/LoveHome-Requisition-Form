  const SUB_CATS = {
    Civil:     ["Foundation","Masonry","Concrete","Waterproofing","Finishing","Roofing","Flooring"],
    Electric:  ["Wiring","Panels & MCB","Lighting","Conduits","Earthing","DB & Fittings","Switches & Sockets"],
    Plumbing:  ["CPVC Pipes","PVC Pipes","UPVC Pipes","Fittings","Valves","Tanks & Pumps","Drainage"],
    Paints:    ["Interior Emulsion","Exterior Emulsion","Primer","Putty","Texture","Enamel","Wood Polish"],
    Furniture: ["Site Furniture","Office Furniture","Fixtures","Doors & Windows","Wardrobes","Hardware Fittings"],
    Hardware:  ["Fasteners","Tools","Locks & Hinges","Glass & Mirrors","Adhesives","Scaffolding"],
    Safety:    ["Helmets","Gloves","Harness","Safety Nets","Signs & Boards","First Aid","Footwear"],
    Office:    ["Stationery","Electronics","Housekeeping","Pantry Supplies","Branding Material","IT Equipment"]
  };

  let reqSeq = 1;

  function genReqNo() {
    return "LH/REQ/" + new Date().getFullYear() + "/" + String(reqSeq).padStart(3, "0");
  }

  document.getElementById("req-no-display").textContent = genReqNo();

  function updateSubCategory() {
    const cat = document.getElementById("item-category").value;
    const sel = document.getElementById("sub-category");
    sel.innerHTML = "";
    if (!cat) { sel.innerHTML = '<option value="">Select category first</option>'; return; }
    sel.innerHTML = '<option value="">Select sub-category</option>';
    (SUB_CATS[cat] || []).forEach(s => {
      const o = document.createElement("option");
      o.value = s; o.textContent = s;
      sel.appendChild(o);
    });
  }

  let rowCount = 0;

  function addSkuRow() {
    rowCount++;
    const n = rowCount;
    const wrap = document.getElementById("sku-rows");
    const row = document.createElement("div");
    row.className = "sku-row";
    row.id = "row-" + n;
    const currentLen = document.querySelectorAll(".sku-row").length;
    row.innerHTML = `
      <span class="row-num">${currentLen + 1}</span>
      <input type="text" class="sku-code" placeholder="SKU e.g. 025-ELB-CPVC" oninput="updateCount()" autocomplete="off">
      <input type="text" class="desc" placeholder="Description (optional)">
      <input type="number" class="qty" placeholder="Qty" min="1" oninput="updateCount()">
      <button class="remove-btn" onclick="removeRow(${n})" title="Remove row">&#215;</button>
    `;
    wrap.appendChild(row);
    updateRowNumbers();
    updateCount();
    notifyResize();
    row.querySelector(".sku-code").focus();
  }

  function removeRow(id) {
    const el = document.getElementById("row-" + id);
    if (el) { el.remove(); updateRowNumbers(); updateCount(); notifyResize(); }
  }

  function updateRowNumbers() {
    document.querySelectorAll(".sku-row").forEach((r, i) => {
      const num = r.querySelector(".row-num");
      if (num) num.textContent = i + 1;
    });
  }

  function updateCount() {
    const rows = document.querySelectorAll(".sku-row");
    let filled = 0;
    rows.forEach(r => {
      const sku = r.querySelector(".sku-code").value.trim();
      const qty = parseInt(r.querySelector(".qty").value);
      if (sku && qty > 0) filled++;
    });
    const el = document.getElementById("sku-count");
    el.innerHTML = "<strong>" + rows.length + "</strong> row" + (rows.length !== 1 ? "s" : "") +
      (filled > 0 ? " &nbsp;&#183;&nbsp; <strong>" + filled + "</strong> complete" : "");
  }

  function showToast(msg, type = "success") {
    const t = document.getElementById("toast");
    t.textContent = msg;
    t.className = "toast" + (type !== "success" ? " " + type : "");
    t.style.display = "block";
    clearTimeout(t._timer);
    t._timer = setTimeout(() => { t.style.display = "none"; }, 4000);
  }

  function resetForm() {
    if (!confirm("Reset all form fields? This cannot be undone.")) return;
    document.getElementById("project").value = "";
    document.getElementById("item-category").value = "";
    updateSubCategory();
    document.getElementById("req-date").value = "";
    document.getElementById("filled-by").value = "";
    document.getElementById("notes").value = "";
    document.getElementById("p-regular").checked = true;
    document.getElementById("sku-rows").innerHTML = "";
    rowCount = 0;
    updateCount();
    document.getElementById("submit-status").textContent = "Fill all required fields (*) to submit";
    addSkuRow();
    notifyResize();
  }

  function collectItems() {
    const rows = document.querySelectorAll(".sku-row");
    const items = [];
    let hasError = false;
    rows.forEach(r => {
      const sku = r.querySelector(".sku-code").value.trim();
      const desc = r.querySelector(".desc").value.trim();
      const qty = parseInt(r.querySelector(".qty").value);
      if (sku && qty > 0) items.push({ sku, desc, qty });
      else if (sku || r.querySelector(".qty").value.trim()) hasError = true;
    });
    return { items, hasError };
  }

  async function submitForm() {
    const project  = document.getElementById("project").value;
    const category = document.getElementById("item-category").value;
    const subCat   = document.getElementById("sub-category").value;
    const reqDate  = document.getElementById("req-date").value;
    const filledBy = document.getElementById("filled-by").value;
    const notes    = document.getElementById("notes").value.trim();
    const priority = document.querySelector('input[name="priority"]:checked')?.value || "Regular";
    const reqNo    = genReqNo();

    const { items, hasError } = collectItems();

    if (!project || !category || !reqDate || !filledBy) {
      showToast("Please fill all required fields (Project, Category, Date, Filled By)", "error");
      return;
    }
    if (items.length === 0) {
      showToast("Add at least one SKU with quantity", "error");
      return;
    }
    if (hasError) {
      showToast("Some SKU rows are incomplete — complete or remove them", "error");
      return;
    }

    const payload = {
      reqNo, project, category,
      subCategory: subCat,
      requiredDate: reqDate,
      filledBy, priority, notes,
      items: items.map(i => i.sku + " × " + i.qty + (i.desc ? " (" + i.desc + ")" : "")).join(" | "),
      itemsJson: JSON.stringify(items),
      timestamp: new Date().toISOString()
    };

    const btn = document.getElementById("submit-btn");
    btn.disabled = true;
    btn.textContent = "Submitting...";
    document.getElementById("submit-status").textContent = "Sending to Google Sheets...";
    showToast("Submitting requisition...", "sending");

    try {
      await fetch(SHEET_URL, {
        method: "POST",
        mode: "no-cors",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify(payload)
      });

      showToast("✓ Requisition " + reqNo + " submitted!");
      document.getElementById("submit-status").textContent = "Submitted: " + reqNo + " at " + new Date().toLocaleTimeString();
      reqSeq++;
      document.getElementById("req-no-display").textContent = genReqNo();

      document.getElementById("sku-rows").innerHTML = "";
      document.getElementById("project").value = "";
      document.getElementById("item-category").value = "";
      updateSubCategory();
      document.getElementById("req-date").value = "";
      document.getElementById("notes").value = "";
      document.getElementById("p-regular").checked = true;
      rowCount = 0;
      addSkuRow();
      updateCount();
      notifyResize();

    } catch (err) {
      showToast("Submission failed. Check your internet connection.", "error");
      document.getElementById("submit-status").textContent = "Submission failed — try again";
    } finally {
      btn.disabled = false;
      btn.textContent = "Submit Requisition";
    }
  }

  // ── Google Sites iframe auto-height ──
  function notifyResize() {
    const h = document.documentElement.scrollHeight;
    try { window.parent.postMessage({ type: 'setHeight', height: h }, '*'); } catch(e) {}
    try { window.parent.postMessage({ gsft_h: h }, '*'); } catch(e) {}
  }

  window.addEventListener('load', notifyResize);
  window.addEventListener('resize', notifyResize);
  new MutationObserver(notifyResize).observe(document.body, { childList: true, subtree: true, attributes: true });

  addSkuRow();
